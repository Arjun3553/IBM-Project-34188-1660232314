import random;
import time;

# while loop for repeated iterations.
while True:
    temperature = str(random.randrange(30,200)) # random values generated by randrange() function.
    humidity = str(random.randrange(0,100)) # random values generated by randrange() function.
    print("temperature : "+ temperature+" °F"+"\n") # to print the temperature in °F.
    print("humidity : "+humidity+" %"+"\n") # to print the humidity in %.
    if(int(temperature) >=100): # if condition to check temperature is high or not.
        print("high temperature : "+ temperature +" °F"+"\n") # to print the temperature if it is high.
    time.sleep(5) # sleep() delays the iteration by 5 seconds.
