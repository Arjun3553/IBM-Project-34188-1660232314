import random;
import time;


while True:
   temperature = random.randint(30,205)
   humidity = random.randrange(0,100)
   print("temperature : "+str(temperature)+" °F")
   print("humidity : "+str(humidity)+" %")
   if(temperature >=100):
    print("high temperature : "+ str(temperature) +" °F")
    print("\n")
   time.sleep(5)
